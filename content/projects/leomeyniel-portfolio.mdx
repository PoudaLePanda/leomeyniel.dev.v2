```typescript
// Utilitaires MDX personnalisés
export async function getMdxFileBySlug(
  directory: 'notes' | 'projects', 
  slug: string
): Promise<MdxFile | null> {
  const filePath = path.join(process.cwd(), 'content', directory, `${slug}.mdx`);
  
  if (!fs.existsSync(filePath)) return null;
  
  const fileContent = fs.readFileSync(filePath, 'utf8');
  const { data: frontmatter } = matter(fileContent);
  
  return {
    slug,
    frontmatter: frontmatter as Frontmatter,
    content: fileContent
  };
}
```

```typescript
// Système de cache intelligent pour MDX
export async function getMdxSlugs(directory: 'notes' | 'projects'): Promise<string[]> {
  const contentDir = path.join(process.cwd(), 'content', directory);
  
  if (!fs.existsSync(contentDir)) {
    return [];
  }
  
  const files = fs.readdirSync(contentDir);
  return files
    .filter(file => file.endsWith('.mdx'))
    .map(file => file.replace(/\.mdx$/, ''));
}
```
